གྲོགས་པོ་ཁྱོད་ཅིའི་ཕྱིར་བྲང་རྡུང་ཞིང་ངུ་ལ་མྱ་ངན་བྱེད་དེ། ངུ་ཞིང་སྨྲེ་སྔགས་འདོན། དེས་སྨྲས་པ། བདག་ནི་འདིར་ལྷའི་བདེ་བ་དང་བྲལ་ཏེ་ཞག་བདུན་ན་འདི་ནས་རྒྱལ་པོའི་ཁབ་ཀྱི་གྲོང་ཁྱེར་ཆེན་པོར་ཕག་མོའི་མངལ་དུ་སྐྱེ་བར་འགྱུར་ཏེ། བདག་ནི་དེར་ལོ་མང་པོ་ཞིག་ཏུ་ཕྱི་ས་དང་གཅིན་འཚལ་བར་འགྱུར་བས་དེའི་སླད་དུ་བདག་དེ་སྐད་མཆིའོ། །

གྲོགས་པོ་ཚུར་ཤོག །ཁྱོད་རྐང་གཉིས་རྣམས་ཀྱི་མཆོག་སངས་རྒྱས་ལ་སྐྱབས་སུ་སོང་ཤིག །འདོད་ཆགས་དང་བྲལ་བ་རྣམས་ཀྱི་མཆོག་ཆོས་ལ་སྐྱབས་སུ་སོང་ཤིག །ཚོགས་རྣམས་ཀྱི་མཆོག་དགེ་འདུན་ལ་སྐྱབས་སུ་སོང་ཤིག །དེ་ནས་ལྷའི་བུ་དེ་དུད་འགྲོའི་སྐྱེ་གནས་ཀྱིས་འཇིགས་ཤིང་སྐྲག །འཆི་བས་འཇིགས་ཤིང་སྐྲག་སྟེ། དེ་འདི་སྐད་ཅེས་མཆིའོ། །

གྲོགས་པོ་བདག་ནི་འདི་ལྟར་རྐང་གཉིས་རྣམས་ཀྱི་མཆོག་སངས་རྒྱས་ལ་སྐྱབས་སུ་མཆིའོ། །

འདོད་ཆགས་དང་བྲལ་བ་རྣམས་ཀྱི་མཆོག་ཆོས་ལ་སྐྱབས་སུ་མཆིའོ། །

ཚོགས་རྣམས་ཀྱི་མཆོག་དགེ་འདུན་ལ་སྐྱབས་སུ་མཆིའོ། །

དེ་ནས་ལྷའི་བུ་དེས་སྐྱབས་གསུམ་བཟུང་ནས་འཆི་འཕོའི་དུས་བགྱིས་པ་དེ། བཅོམ་ལྡན་འདས་གང་དུ་སྐྱེས་ལགས། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། ཀཽ་ཤི་ཀ་ལྷའི་བུ་དེ་ནི་དགའ་ལྡན་ལྷའི་རིས་དེར་སྐྱེས་ཏེ། ལྷའི་འདོད་པ་ཐམས་ཅད་ཀྱང་ཕུན་སུམ་ཚོགས་སོ། །

དེ་ནས་ལྷའི་དབང་པོ་བརྒྱ་བྱིན་ཚིམ་ཞིང་མགུ་སྟེ། རབ་ཏུ་དགའ་ནས་དགའ་བ་དང་ཡིད་བདེ་བ་སྐྱེས་ཏེ། དབང་པོ་ཚིམ་ནས་དེའི་ཚེ་ཚིགས་སུ་བཅད་དེ་གསོལ་པ། གང་དག་སངས་རྒྱས་སྐྱབས་མཆི་བ། །

དེ་དག་ངན་འགྲོར་མི་མཆི་སྟེ། །

མི་ཡི་ལུས་རྣམས་སྤངས་ནས་སུ། །

དེ་དག་ལྷ་ལུས་ཐོབ་པར་འགྱུར། །

གང་དག་ཆོས་ལ་སྐྱབས་མཆི་བ། །

དེ་དག་ངན་འགྲོར་མི་མཆི་སྟེ། །

མི་ཡི་ལུས་རྣམས་སྤངས་ནས་སུ། །

དེ་དག་ལྷ་ལུས་ཐོབ་པར་འགྱུར། །

གང་དག་དགེ་འདུན་སྐྱབས་མཆི་བ། །

དེ་དག་ངན་འགྲོར་མི་མཆི་སྟེ། །

མི་ཡི་ལུས་རྣམས་སྤངས་ནས་སུ། །

དེ་དག་ལྷ་ལུས་ཐོབ་པར་འགྱུར། །

གང་དག་ཉིན་མཚན་རྟག་པར་ཡང་། །

སངས་རྒྱས་རྗེས་སུ་དྲན་པ་དང་། །

གང་དག་སངས་རྒྱས་སྐྱབས་མཆིས་པ། །

མི་དེ་དག་གི་རྙེད་པ་ཆེ། །

གང་དག་ཉིན་མཚན་རྟག་པར་ཡང་། །

ཆོས་ནི་རྗེས་སུ་དྲན་པ་དང་། །

གང་དག་ཆོས་ལ་སྐྱབས་མཆིས་པ། །

མི་དེ་དག་གི་རྙེད་པ་ཆེ། །

གང་དག་ཉིན་མཚན་རྟག་པར་ཡང་། །

དགེ་འདུན་རྗེས་སུ་དྲན་པ་དང་། །

མི་དེ་དག་གི་རྙེད་པ་ཆེ། །

དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ལྷའི་དབང་པོ་བརྒྱ་བྱིན་གྱི་རྗེས་སུ་འཐུན་པར་བརྗོད་པའི་ཚིགས་སུ་བཅད་པ་བཀའ་སྩལ་པ། ཀཽ་ཤི་ཀ་དེ་དེ་བཞིན་ཏེ། དེ་དེ་བཞིན་ནོ། །

གང་དག་སངས་རྒྱས་སྐྱབས་དོང་བ། །

དེ་དག་ངན་འགྲོར་མི་འགྲོ་སྟེ། །

དེ་དག་ངན་འགྲོར་མི་འགྲོ་སྟེ། །

མི་ཡི་ལུས་རྣམས་སྤངས་ནས་སུ། །

དེ་དག་ལྷ་ལུས་ཐོབ་པར་འགྱུར། །

གང་དག་དགེ་འདུན་སྐྱབས་དོང་བ། །

དེ་དག་ངན་འགྲོར་མི་འགྲོ་སྟེ། །

མི་ཡི་ལུས་རྣམས་སྤངས་ནས་སུ། །

དེ་དག་ལྷ་ལུས་ཐོབ་པར་འགྱུར། །

དེ་ནས་ལྷའི་དབང་པོ་བརྒྱ་བྱིན་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་དགའ་ནས་བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། བཅོམ་ལྡན་འདས་ལ་ལག་པ་ཐལ་མོ་སྦྱར་ནས་ལན་གསུམ་བསྐོར་བ་བྱས་ཏེ། དེ་ཉིད་དུ་མི་སྣང་བར་གྱུར་ཏོ། །

བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས་དགེ་སློང་དེ་དག་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ། །
