རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཏ་ཐཱ་ག་ཏ་ཛྙཱ་ན་མུ་དྲཱ་ས་མཱ་དྷི་ནཱ་མ་མ་ཧཱ་ཡཱ་ན་སཱུ་ཏྲ། བོད་སྐད་དུ། འཕགས་པ་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱའི་ཏིང་ངེ་འཛིན་ཅེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ། བམ་པོ་དང་པོ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །

འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་ན་འོད་མའི་ཚལ་བྱ་ཀ་ལན་ད་ཀའི་གནས་ན། དགེ་སློང་སྟོང་ཉིས་བརྒྱ་ལྔ་བཅུའི་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་དང་། བྱང་ཆུབ་སེམས་དཔའ་བྱེ་བ་ཕྲག་སུམ་ཅུ་ཐམས་ཅད་ཀྱང་གཟུངས་ཐོབ་པ། སྤོབས་པ་ཐོགས་པ་མེད་པ། ཏིང་ངེ་འཛིན་ཐོབ་པ། སྟོང་པ་ཉིད་ཀྱི་ཏིང་ངེ་འཛིན་ལ་གནས་པ། མཚན་མ་མེད་པའི་སྤྱོད་ཡུལ་པ། སྨོན་ལམ་བཅོས་མ་མ་ཡིན་པ། སྨོན་པ་མེད་པ་ལ་གནས་པ། སྤྱོད་ལམ་བཅོས་པ་མེད་པ། སྒོ་མཐའ་ཡས་པ་སྒྲུབ་པའི་གཟུངས་ཐོབ་པ། མཉམ་པ་ཉིད་དུ་བཟོད་པ་ཐོབ་པ། མཉམ་པ་ཉིད་ཐོབ་པ། ཐོགས་པ་མེད་པའི་གཟུངས་ཐོབ་པ། སེམས་ཅན་ཐམས་ཅད་ཀྱི་དབང་པོ་མཆོག་དང་མཆོག་མ་ཡིན་པ་ཤེས་པ་དང་། བསམ་པ་དང་། སྤྱོད་པའི་ཚུལ་ཤིན་ཏུ་ཤེས་པ་ཤ་སྟག་དང་ཐབས་ཅིག་ཏུ་བཞུགས་ཏེ། དེ་ནས་དེའི་ཚེ་བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པའི་བྱིན་གྱི་རླབས་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཡུལ་ལ་འཇུག་པའི་ཏིང་ངེ་འཛིན་གཟུགས་མེད་པ། བསྟན་དུ་མེད་པ། གཟུང་བ་མེད་པ། ཐོགས་པ་མེད་པ། སྣང་བ་མེད་པ། རྣམ་པར་རིག་པ་མེད་པ། རྣམ་པར་རིག་པར་བྱ་བ་མ་ཡིན་པ། སེམས་དང་ཡིད་དང་རྣམ་པར་ཤེས་པ་དང་བྲལ་བ། རྩ་བ་མེད་པ། རྟེན་པ་མེད་པ། དམིགས་སུ་མེད་པ། མི་མཉམ་པ་དང་མཉམ་པ། མཉམ་པ་མེད་པ། ཟླ་མེད་པ། ང་ཡི་བ་མེད་པ། ཡོངས་སུ་འཛིན་པ་མེད་པ། མི་བྱ་བ་མེད་པ། མ་བྱས་པ། མ་སོང་བ། མ་འོངས་པ། འགྲོ་བ་མེད་པ། མི་གནས་པ། རྒྱུན་ཆད་པ་མེད་པ། གནས་མེད་པ། མཚན་མ་མེད་པ། མཚན་མ་ཐམས་ཅད་དང་བྲལ་བ། བླང་བ་མེད་པ། དོར་བ་མེད་པ། འདུས་མ་བྱས་པ། མངོན་པར་འདུ་མི་བྱེད་པ། འདུ་བ་མེད་པ། བྲལ་བ་མེད་པ། སེམས་མེད་པ། སེམས་དང་བྲལ་བ། ཀུན་ཏུ་འབྱུང་བ་མེད་པ། རྩོམ་པ་མེད་པ། ཡོངས་སུ་གྲུབ་པ་མེད་པ། མ་བྱུང་བ། ཀུན་ཏུ་མ་བྱུང་བ། རབ་ཏུ་རྒྱུ་བ་མེད་པ། ཡང་དག་པར་རྒྱུ་བ་མེད་པ། ཤིན་ཏུ་སྟོང་པ། འདྲེ་བ་མེད་པ། ཆོས་ཐམས་ཅད་ལ་སྒྲོ་བཏགས་པ་མེད་པ་འདི་ལྟ་སྟེ། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཡུལ་ལ་འཇུག་པའི་གཟུངས་ཀྱི་སྒོ་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ལ་སྙོམས་པར་ཞུགས་སོ། །

ཏིང་ངེ་འཛིན་དེ་ལ་སྙོམས་པར་ཞུགས་མ་ཐག་ཏུ་བཅོམ་ལྡན་འདས་ཀྱི་སྐུ་ཡང་མི་སྣང་སྐུའི་མཚན་མ་ཡང་མི་སྣང་། སེམས་དང་སེམས་ཀྱི་མཚན་མ་ཡང་མི་སྣང་། ན་བཟའ་དང་ན་བཟའི་མཚན་མ་ཡང་མི་སྣང་། གདན་དང་གདན་གྱི་མཚན་མ་ཡང་མི་སྣང་། གཟིམས་པ་དང་གཟིམས་པའི་མཚན་མ་ཡང་མི་སྣང་། འཆག་པ་དང་འཆག་པའི་མཚན་མ་ཡང་མི་སྣང་། བཞེངས་པ་དང་བཞེངས་པའི་མཚན་མ་ཡང་མི་སྣང་། གནས་པ་དང་གནས་པའི་མཚན་མ་ཡང་མི་སྣང་། གྲིབ་མའི་དབྱིབས་དང་གྲིབ་མའི་དབྱིབས་ཀྱི་མཚན་མ་ཡང་མི་སྣང་། བཞུགས་པ་དང་བཞུགས་པའི་མཚན་མ་ཡང་མི་སྣང་ཞིང་སྒྲ་དང་སྒྲའི་མཚན་མ་ཡང་མི་གྲག་གོ། །

དེ་ནས་ཏིང་ངེ་འཛིན་དེའི་མཐུ་དང་། སངས་རྒྱས་ཀྱི་བྱིན་གྱི་རླབས་ཀྱིས་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་འཇིག་རྟེན་གྱི་ཁམས་འདི་སྣང་བ་ཆེན་པོས་ཁྱབ་པར་གྱུར་ནས། འདི་ལྟ་བུའི་འོད་ཀྱིས་སྣང་བར་བྱས་ཏེ། འོད་དེས་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་འཇིག་རྟེན་གྱི་ཁམས་འདི་ན་ཟླ་བའི་འོད་ཀྱང་མི་སྣང་། ཉི་མ་དང་། སྐར་མ་དང་། མེ་དང་། ནོར་བུ་དང་། གློག་གི་འོད་རྣམས་ཀྱང་མི་སྣང་། དྲི་ཟ་རྣམས་དང་། མིའམ་ཅི་རྣམས་དང་། འདོད་པ་ན་སྤྱོད་པ་དང་། གཟུགས་ན་སྤྱོད་པའི་ལྷ་རྣམས་དང་། སྐར་མ་རྣམས་དང་། ལྷའི་གཞལ་མེད་ཁང་གི་འོད་ཀྱང་མི་སྣང་། བརྒྱ་བྱིན་དང་། ཚངས་པ་དང་། འཇིག་རྟེན་སྐྱོང་བ་རྣམས་ཀྱི་འོད་ཀྱང་མི་སྣང་ངོ་། །

མི་གསལ་བ། མ་རེག་པ། འདོད་པར་བྱ་བ་མ་ཡིན་པ། འོད་མེད་པ་མི་སྣང་བ། གཟུང་བ་མེད་པ། སྣང་བ་མེད་པ། མུན་པ་མེད་པ། མུན་པ་མ་ཡིན་པ། མུན་པ་དང་བྲལ་བ། གསལ་བ་མ་ཡིན་པ། གནས་མེད་པ། གནས་དང་བྲལ་བ། མ་ཞི་བ། རབ་ཏུ་མ་ཞི་བ། ཞི་བ་མེད་པ། ཞི་བ་དང་བྲལ་བ། ཞི་བ་མཉམ་པ། དག་པ། རྣམ་པར་དག་པ། ཡོངས་སུ་དག་པ། ཅི་ཡང་མེད་པ། ལེན་པ་མེད་པ། ཉེ་བར་ལེན་པ་མེད་པ། མཛའ་བ་མེད་པ། འཐབ་པ་མེད་པ། འཐབ་པ་དང་བྲལ་བ། མི་གནས་པའི་ཚུལ་གྱིས་གནས་པ། མདུད་པ་མེད་པ། མདུད་པ་དང་བྲལ་བ། དང་བ། རྙོག་པ་མེད་པ། ཤི་འཕོ་མེད་པ། ཤི་འཕོར་གྱུར་པ་མེད་པ། ཆོས་མེད་པ། ཆོས་མ་ཡིན་པ་མེད་པ། ཞིང་མེད་པ། ཞིང་མ་ཡིན་པ་མེད་པ། མི་ཟད་པ། ཟད་པར་བྱ་བ་མ་ཡིན་པ། ཟད་པ་དང་བྲལ་བ། རྡུལ་མེད་པ། སྐད་ཅིག་པ། མི་འཁྲུགས་པ། ཡི་གེ་ཡོངས་སུ་སྤངས་པ། སྒྲ་ཡོངས་སུ་སྤངས་པ། དབྱངས་ཡོངས་སུ་སྤངས་པ། བཤད་པ་ཡོངས་སུ་སྤངས་པ། བསྒོམ་པ་ཡོངས་སུ་སྤངས་པ། སེམས་པ་ཡོངས་སུ་སྤངས་པ། སེམས་ཡོངས་སུ་སྤངས་པ། རིགས་པ་མེད་པ། རིགས་པ་མེད་པ་མ་ཡིན་པ། རིགས་པ་དང་བྲལ་བ་མ་ཡིན་པ། ལྡན་པ་མ་ཡིན་པ། ལྡན་པ་མེད་པ་མ་ཡིན་པ། མི་ལྡན་པ་མ་ཡིན་པ། ཟད་པ་མེད་པ། ཟད་པ་མེད་པ་མ་ཡིན་པ། འཆི་བ་མེད་པ། འཆི་བ་མེད་པ་མ་ཡིན་པ། ཞི་བར་བྱ་བ་མ་ཡིན་པ། མཚན་མ་མེད་པ། མཚན་མ་མེད་པ་མ་ཡིན་པ། དཔག་ཏུ་མེད་པ། དཔག་ཏུ་མེད་པ་མ་ཡིན་པ། དཔག་པར་བྱ་བ་མ་ཡིན་པ། མ་སོང་བ། མ་འོངས་པ། གཉིས་སུ་མེད་པ། གཉིས་སུ་མེད་པ་མ་ཡིན་པ། དམིགས་པ་མེད་པ། ཚུ་རོལ་མ་ཡིན་པ། ཕ་རོལ་མ་ཡིན་པ། གཞི་མེད་པ། གནས་མེད་པ། མི་འབྱུང་བ། མི་འཇིག་པ། མ་འཕོས་པ། མཉན་དུ་མེད་པ། ཕ་རོལ་གནོན་པ་མེད་པ། ལེན་པ་མེད་པ། འདོར་བ་མེད་པ། མཚན་ཉིད་མེད་པ། མཚན་ཉིད་དང་བྲལ་བ། མཚན་ཉིད་མ་ཡིན་པ། མཚན་ཉིད་ཡོངས་སུ་སྤངས་པ། ལས་མེད་པ། ལས་མེད་པ་མ་ཡིན་པ། མཚན་མེད་པ། མཚན་མེད་པ་མ་ཡིན་པ། མཚན་གྱིས་བརྒྱན་པ། སྒོ་མེད་པ། མི་ཆགས་པ། ཆགས་པ་དང་བྲལ་བ། དད་པར་བྱེད་པ། སྦྱར་བ་མེད་པ། སྦྱར་བ་དང་བྲལ་བ། བདེ་བར་བྱེད་པ། མཆོག་ཏུ་བདེ་བར་བྱེད་པ། སྐྱེ་མཆེད་དྲུག་ལས་གྲོལ་བ། མི་སྣང་བ། དེ་བཞིན་ཉིད་ཀྱིས་རབ་ཏུ་ཕྱེ་བ་དེ་ཁོ་ན་ཉིད། དེ་བཞིན་ཉིད་སེམས་ཅན་རྣམས་མི་སྒྲོལ་བའི་ཚུལ་གྱིས་སྒྲོལ་བ། སེམས་ཅན་རྣམས་མི་འགྲོལ་བའི་ཚུལ་གྱིས་འགྲོལ་བ། སེམས་ཅན་རྣམས་ཡོངས་སུ་མི་སྦྱོང་བའི་ཚུལ་གྱིས་སྦྱོང་བ། སེམས་ཅན་རྣམས་ལ་མི་སྐྱོབ་པའི་ཚུལ་གྱིས་སྐྱོབ་པ། གཉིས་སུ་མེད་པའི་ཚུལ་གྱིས་སེམས་ཅན་རྣམས་ལ་ཆོས་ཀྱང་སྟོན་པ། མཉམ་པ་མེད་པ། མི་མཉམ་པ་དང་། མཉམ་པ། འདྲ་བ་མེད་པ། ཟླ་མེད་པ། མི་མཆི་བ་དང་མཉམ་པ། ནམ་མཁའ་དང་མཉམ་པ། གནས་མེད་པ་དང་མཉམ་པ། ཤིན་ཏུ་མཉམ་པ། མ་སྐྱེས་པ། སྐྱེ་བ་མེད་པ་དང་མཉམ་པ། མི་དམིགས་པ། དམིགས་པ་མེད་པ་དང་མཉམ་པ། མ་ཞི་བ། ཞི་བ་དང་བྲལ་བ། ཞི་བ་ཉམས་པ། ཞི་བ་རབ་ཏུ་ཞི་བ། ཉེ་བར་ཞི་བ། འཇམ་པ། མཉེན་པ། ཤིན་ཏུ་དྲི་མ་མེད་པ། ཤིན་ཏུ་དུལ་བ། རྟག་པ་མ་ཡིན་པ། ཆད་པ་མ་ཡིན་པ། སྤྱོད་པ་ལས་ལྡོག་པ། ལྡོག་པ་མེད་པ་དེས་པ། འཇིགས་པ་ལས་ལྡོག་པ། ཕྱིར་མི་ལྡོག་པ། འཛིན་པ་ཐམས་ཅད་གཅོད་པ། མདུད་པ་ཐམས་ཅད་གཅོད་པ། གསལ་བ། ཐ་དད་པ་མེད་པ། གཉིས་སུ་མེད་པ། གཉིས་དང་བྲལ་བ། གཉིས་སུ་མེད་པར་ཡོངས་སུ་བསྟན་པ། མ་བྱས་པའི་ངོ་བོ་ཉིད་རང་བཞིན་གྱིས་འོད་གསལ་བ་ཤིན་ཏུ་ཀུན་ཏུ་འབྱུང་བ་མེད་པ། རྣམ་པར་མ་གྲོལ་བའི་ཚུལ་གྱིས་སྤྱོད་པ་ལས་རྣམ་པར་གྲོལ་བ། འདུ་བ་མེད་པ། བྲལ་བ་མེད་པ། རིང་པོ་མ་ཡིན་པ། ཐུང་ངུ་མ་ཡིན་པ། ཟླུམ་པོ་མ་ཡིན་པ། གྲུ་བཞི་མ་ཡིན་པ། ཀུན་ཏུ་དཀྱིལ་འཁོར་མ་ཡིན་པ། སྦོམ་པོ་མ་ཡིན་པ། ཕུང་པོས་རབ་ཏུ་ཕྱེ་བ་མ་ཡིན་པ། ཁམས་ཀྱིས་རབ་ཏུ་ཕྱེ་བ་མ་ཡིན་པ། སྐྱེ་མཆེད་ཀྱིས་རབ་ཏུ་ཕྱེ་བ་མ་ཡིན་པ། འདུས་བྱས་སུ་རྣམ་པར་བརྟག་པ་མ་ཡིན་པ། འདུས་མ་བྱས་ལས་བྱུང་བ་མ་ཡིན་པ། འཕེལ་བ་མེད་པ། འབྲི་བ་མེད་པ། སོགས་པ་མེད་པ། མ་སྐྱེས་པ། སུས་ཀྱང་མ་མཐོང་བ། མ་ཐོས་པ། རྣམ་པར་མ་ཤེས་པ། ཡང་དག་པ་མ་ཡིན་པ། ཡང་དག་པ་མ་ཡིན་པའང་མ་ཡིན་པ། མ་སྐྱེས་པ། འཆི་འཕོ་མ་ཡིན་པ། འཇིག་རྟེན་དང་ལྷན་ཅིག་པ་མ་ཡིན་པ། ལྷན་ཅིག་པ་མ་ཡིན་པ་འང་མ་ཡིན་པ་སྟེ། ལུས་དེ་ནི་གཡོ་བ་མེད་པ། འཁྲུལ་པ་མེད་པ། རྒོད་པ་མེད་པ། རྒྱུ་བ་མེད་པ། རྣམ་པར་རྒྱུ་བ་མེད་པ། བསྒོམ་པ་མེད་པ། བསྒོམ་པ་དང་བྲལ་བ་མ་ཡིན་པ། འབྱུང་བ་མ་ཡིན་པ། འཇིག་པ་མ་ཡིན་པ། ངོ་བོ་ཉིད་མ་ཡིན་པ། ངོ་བོ་ཉིད་མེད་པའང་མ་ཡིན་པ། སེམས་པ་མེད་པ། སྟོན་པ་མེད་པ། སྦྱོར་བ་མེད་པ། མི་སྦྱོར་བ། མི་སྦྱོར་བའང་མ་ཡིན་པ། འདོད་པ་ལ་འདོད་ཆགས་པ་མ་ཡིན་པ། འདོད་པ་ལ་འདོད་ཆགས་མེད་པ་མ་ཡིན་པ། མེད་པ་མེད་པ་མ་ཡིན་པ། མཚན་ཉིད་མེད་པ། མཚན་ཉིད་མེད་པ་མ་ཡིན་པ། མྱ་ངན་ལས་འདའ་བ་མེད་པ། ཡོངས་སུ་མྱ་ངན་ལས་མ་འདས་པ། གནས་པ་མེད་པ། གནས་པ་མེད་པ་མ་ཡིན་པ་སྟེ། དབྱངས་ཀྱི་ཡན་ལག་ཀུན་དང་སྣ་ཚོགས་མཐའ་ཡས་སྐད་རྣམས་ཡོངས་སུ་སྦྱོང་། །

རྗེས་སུ་རྟོགས་དང་ཚིམ་པ་དང་ནི་སྤྱོད་དང་འཛིན་པ་ཐམས་ཅད་བྲལ། །

ཤིན་ཏུ་རྣམ་པར་ཤེས་དང་གདོན་མི་ཟ་མཆོག་ཉོན་མོངས་ཐམས་ཅད་ཟད། །

མདོ་སྡེ་འདི་ལ་རབ་ཏུ་སྦྱོར་བ་བྱང་ཆུབ་དྲུང་དུ་སོང་བ་ཡིན། །

རྩེ་གཅིག་ཡིད་ཀྱིས་མངོན་སུམ་དུ་ནི་འདི་ལ་ཚུལ་བཞིན་རབ་སྦྱོར་ལ། །

གཉིད་དང་ལེ་ལོ་མ་བྱེད་ཤེས་ཀྱི་ཁེ་བདེ་མངོན་པར་དགའ་མ་བྱེད། །

རྟག་ཏུ་ཚུལ་ཁྲིམས་གཉེན་དང་ཕྲག་དོག་མེད་དང་བྱམས་པའི་ཡིད་དུ་བྱོས། །

དགའ་བའི་ངང་ཚུལ་གུས་དང་བཅས་པས་ཏིང་འཛིན་མཆོག་འདི་རྙེད་པར་འགྱུར། །

མཉམ་སེམས་ཤིན་ཏུ་བསྡམས་པ་དྲང་པོ་རྟག་ཏུ་རྣམ་པར་དགའ་བ་དང་། གནས་མེད་འདྲིས་མེད་ཤེས་ཀྱི་ཁེ་དང་བདེ་ལ་པད་མ་ཆུ་བཞིན་བྱོས། །

ཇི་ལྟར་བྱ་བཞིན་ང་མེད་མོས་པ་བརྟན་ཞིང་སྲེད་དང་གཡོ་མེད་པ། །

དེ་ནི་ཕྱི་མའི་དུས་ན་གཟུངས་མཆོག་རྒྱལ་པོ་འདི་ནི་འཐོབ་པར་འགྱུར། །

སྟོང་གསུམ་དག་ནི་རབ་ཏུ་གཡོས་ནས་ལྷ་བརྒྱ་སིལ་སྙན་དཀྲོལ་བར་བྱེད། །

ནམ་མཁའ་ལས་ནི་བྱུག་པ་གོས་དང་ཕྱེ་མ་དག་དང་རྒྱལ་མཚན་བརྒྱ། །

མཛེས་ཕྲེང་མགུལ་རྒྱན་རྣ་ཆ་ཟླ་གམ་དུང་དང་རྐང་གདུབ་དོ་ཤལ་བརྒྱ། །

ནོར་བུ་མུ་ཏིག་དཔུང་རྒྱན་གཙུག་རྒྱན་སོར་གདུབ་སྣ་ཚོགས་འཐོར་བར་བྱེད། །

ལྷ་དང་ལྷ་ཡི་བུ་མོ་ཀླུ་དང་དྲི་ཟ་འདབ་བཟངས་ཚོགས་མང་པོ། །

ནམ་མཁའ་ལྡིང་དང་ལྷ་མིན་དབང་པོ་དགེ་བསྙེན་དགེ་སློང་བརྒྱ་མང་དང་། །

དེ་བཞིན་དགེ་སློང་མ་དང་དགེ་བསྙེན་མ་ནི་མང་པོ་དགའ་འགྱུར་ལངས། །

རང་གི་གོས་དག་དང་ནི་རྒྱན་རྣམས་ཕུལ་ནས་བྱང་ཆུབ་སེམས་བསྐྱེད་དོ། །

ཆོས་འདི་ཤིན་ཏུ་བསྟན་ཚེ་སེམས་ཅན་མང་པོ་བྱང་ཆུབ་ཞུགས་པ་དང་། །

ཁ་ཅིག་ཕྱིར་མི་ལྡོག་པར་གྱུར་པ་བསྟན་ཅིང་ཚད་གཟུང་ནུས་མ་ཡིན། །

དགྲ་བཅོམ་ཉིད་དུ་བཏུལ་བར་གྱུར་པ་མཐའ་ཡས་གང་གཱའི་བྱེ་མ་སྙེད། །

འདྲེན་པ་བལྟ་ཕྱིར་ཞིང་ནི་བརྒྱ་ནས་ལྷགས་པ་དེ་དག་དགའ་ནས་དོང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཆོས་ཀྱི་རྣམ་གྲངས་འདི་བཤད་པ་ན། སེམས་ཅན་གྲངས་མེད་པ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་ཡང་དག་པར་ཞུགས་སོ། །

བྱང་ཆུབ་སེམས་དཔའ་ཁྲག་ཁྲིག་ཕྲག་བརྒྱད་ཅུ་ནི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ལས་ཕྱིར་མི་ལྡོག་པར་གྱུར་ཏོ། །

བྱང་ཆུབ་སེམས་དཔའ་བྱེ་བ་ཕྲག་དྲུག་ཅུ་དང་། འབུམ་ཕྲག་གསུམ་ནི་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱའི་ཏིང་ངེ་འཛིན་འདི་ཐོབ་བོ། །

དགེ་སློང་དང་། དགེ་སློང་མ་དང་། དགེ་བསྙེན་དང་། དགེ་བསྙེན་མ་དཔག་ཏུ་མེད་པ་དང་། སེམས་ཅན་ཚད་མེད་པ་དག་གིས་ནི་དགྲ་བཅོམ་པ་ཉིད་ཐོབ་བོ། །

ཕྱོགས་བཅུའི་འཇིག་རྟེན་གྱི་ཁམས་ནས་འདུས་པའི་བྱང་ཆུབ་སེམས་དཔའ་དེ་དག་ཐམས་ཅད་ཀྱིས་ནི་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱའི་ཏིང་ངེ་འཛིན་འདི་ཐོབ་བོ། །

བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། ཉན་ཐོས་ཆེན་པོ་དེ་དག་དང་། མཆོག་ཏུ་དགའ་བའི་རྒྱལ་པོ་དང་། འཇམ་དཔལ་ལ་སོགས་པ་བྱང་ཆུབ་སེམས་དཔའ་དཔེ་མེད་དྲུག་ཅུ་དང་། བྱམས་པ་དང་། སེང་གེ་རབ་སྣང་དང་སྣང་བྱེད་ལ་ཐུག་པའི་བར་ལ་སོགས་པ་བསྐལ་པ་བཟང་པོའི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཐམས་ཅད་དང་། ཕྱོགས་བཅུའི་འཇིག་རྟེན་གྱི་ཁམས་ནས་འདུས་པའི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་དང་། རྒྱལ་པོའི་བཙུན་མོ་དཔལ་བཟངས་དང་། གསེར་མཆོག་སྣང་དང་། ཐམས་ཅད་དང་ལྡན་པའི་འཁོར་དེ་དང་། ལྷ་དང་། མི་དང་། ལྷ་མ་ཡིན་དང་། དྲི་ཟར་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ།། །

།འཕགས་པ་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱའི་ཏིང་ངེ་འཛིན་ཅེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ་རྫོགས་སོ།། །

།རྒྱ་གར་གྱི་མཁན་པོ་ཛི་ན་མི་ཏྲ་དང་། མུ་ནི་ཝརྨ་དང་། དཱ་ན་ཤཱི་ལ་དང་། ཞུ་ཆེན་གྱི་ལོཙྪ་བ་བན་དེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་སྐད་གསར་ཆད་ཀྱིས་ཀྱང་བཅོས་ནས་གཏན་ལ་ཕབ་པ།།
